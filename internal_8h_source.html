00010     int firstcol = d && (d)->vector ? -1 : 0; \
00011     int vsize = d && (d)->vector ? (d)->vector->size : 0; \
00012     int wsize = d && (d)->weights ? (d)->weights->size : 0; \
00013     int msize1 = d && (d)->matrix ? (d)->matrix->size1 : 0; \
00014     int msize2 = d && (d)->matrix ? (d)->matrix->size2 : 0; \
00015     int tsize = vsize + msize1*msize2; \
